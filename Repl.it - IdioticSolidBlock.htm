#include <stdio.h>
int *largest(int a[], int n){
  int *greatest = a;
  for(int i = 0; i < n; i++){
    if(*greatest>*(a+i)) greatest = a+i;
  }
  return greatest;
}
int main(void) {
  int t[5] = {1,2,3,4,5};
  printf("%d\n", largest(t, 5));
 
}
